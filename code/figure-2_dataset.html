Define functions

plot_data <- function(df = df_plot, type = data_type) {
  
  df %>%
    filter(id == type) %>% 
    ggplot(aes(x = x, y = y, color = labels, shape = labels)) +
    geom_point(show.legend = F) +
    ggtitle(type) +
    theme_bw() +
    theme(
      axis.text.x = element_blank(),
      axis.text.y = element_blank(),
      axis.ticks = element_blank(),
      axis.title = element_blank(),
      panel.border = element_blank(),
      panel.grid = element_blank(),
      panel.spacing = unit(2, "lines"),
      plot.title = element_text(hjust = 0.5, vjust = 3),
      strip.background = element_blank(),
      # strip.text.x = element_blank(),
      text = element_text(
        color = "grey20",
        face = "bold",
        family = "Helvetica",
        size = 12
      ),
    )
}

Create datasets

The code for creating the data was taken from the GitHub of TensorFlow Playground here and is licensed under Apache 2.0.

# Define seed
set.seed(2253)

# Define parameters (noise has to be between 0 and 0.5)
noise <- 0.1
numSamples <- 200

# Create data
playground <- list(
  XOR = classifyXORData(numSamples, noise),
  Spiral = classifySpiralData(numSamples, noise),
  Circle = classifyCircleData(numSamples, noise),
  Gaussian = classifyTwoGaussData(numSamples, noise)
)

# Data for plotting
df_plot <- playground %>% 
  map_dfr(as_tibble) %>% 
  setNames(c("x", "y", "labels")) %>%
  mutate(
    id = rep(names(playground), each = numSamples),
    labels = factor(labels)
  )
